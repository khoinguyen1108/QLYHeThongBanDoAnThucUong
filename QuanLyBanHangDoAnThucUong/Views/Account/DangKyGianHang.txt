<script>
    function showDieuLe() {
        var dieuLeSelect = document.getElementById("MaDieuLe");
        var selectedValue = dieuLeSelect.options[dieuLeSelect.selectedIndex];

        if (selectedValue.value) {
            // Có thể gọi API để lấy chi tiết điều lệ
            document.getElementById("dieuLeContent").innerHTML = selectedValue.text;
            document.getElementById("dieuLeContent").style.display = "block";
        } else {
            document.getElementById("dieuLeContent").style.display = "none";
        }
    }
</script>
